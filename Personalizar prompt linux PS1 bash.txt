prompt_status() {
    local last_exit_code="$?"
    local status_color status_symbol
    # Determina a cor e o s√≠mbolo baseados no resultado do √∫ltimo comando executado.
    if [[ $last_exit_code -eq 0 ]]; then
        status_color="\[\e[1;32m\]" # Verde para indicar sucesso.
        status_symbol="‚àö"
    else
        status_color="\[\e[48;5;95;38;5;214m\]" # Cor personalizada para erro, com fundo e texto coloridos.
        status_symbol="üö©:$last_exit_code"
    fi

    # Constr√≥i a parte din√¢mica do PS1, mostrando o status do √∫ltimo comando.
    PS1="${status_color}${status_symbol}\[\e[0m\] "
}

# Define a configura√ß√£o de cores e s√≠mbolos do usu√°rio para o prompt.
user_prompt_customization() {
    local user_color user_symbol path_color prompt_symbol

    if [[ $EUID -eq 0 ]]; then
        # Configura√ß√µes para o usu√°rio root
        user_color="\[\e[1;31m\]" # Vermelho para o nome de usu√°rio root.
        user_symbol='üíÄ' # S√≠mbolo de caveira para root.
        path_color="\[\e[1;93m\]" # Caminho do diret√≥rio em amarelo para root.
        prompt_symbol='#'
    else
        # Configura√ß√µes para usu√°rios normais
        user_color="\[\e[1;92m\]" # Verde para o nome de usu√°rio normal.
        user_symbol='üêß' # S√≠mbolo de pinguim para usu√°rios normais.
        path_color="\[\e[1;96m\]" # Caminho do diret√≥rio em ciano para usu√°rios normais.
        prompt_symbol='$'
    fi

    # Adiciona a cor do usu√°rio, s√≠mbolo, nome do host, cor do caminho e s√≠mbolo do prompt ao PS1.
    PS1+="${user_color}\u${user_symbol} \h ${path_color}\w \n‚îî‚îÄ${prompt_symbol} \[\e[0m\]"
}

# Inicializa o prompt com status e customiza√ß√£o do usu√°rio.
prompt_init() {
    prompt_status
    user_prompt_customization
}

# Atualiza o prompt antes de exibir cada novo prompt.
PROMPT_COMMAND=prompt_init
